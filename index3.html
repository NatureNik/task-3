const readline = require('readline'); const rl = readline.createInterface({ input: process.stdin, output: process.stdout }); rl.question('Введите число: ', (number) => { if (number == 5 || number == 10) { console.log('Число либо равно 5, либо равно 10'); } else { console.log('Неизвестное число'); } rl.close(); });
